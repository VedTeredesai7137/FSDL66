// app.component.ts import { Component } from '@angular/core'; interface CartItem { name: string; price: number; quantity: number; } @Component({ selector: 'app-root', templateUrl: './app.component.html', styleUrls: ['./app.component.css'] }) export class AppComponent { cart: CartItem[] = []; itemName: string = ''; itemPrice: number | null = null; itemQuantity: number = 1; addItem() { if (this.itemName && this.itemPrice !== null && this.itemPrice > 0) { const existingItem = this.cart.find(item => item.name === this.itemName); if (existingItem) { existingItem.quantity += this.itemQuantity; // Increase quantity if item already in cart } else { this.cart.push({ name: this.itemName, price: this.itemPrice, quantity: this.itemQuantity }); } this.itemName = ''; this.itemPrice = null; this.itemQuantity = 1; } } removeItem(item: CartItem) { this.cart = this.cart.filter(cartItem => cartItem !== item); } getTotalPrice(): number { return this.cart.reduce((total, item) => total + item.price * item.quantity, 0); } }
